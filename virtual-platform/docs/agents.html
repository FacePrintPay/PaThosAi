Usage & Deployment

Quick Start

from pathos import run

args = {
    'function': 'tritonvirtualassistant_development',
    'input': 'path/to/input',
    'output': 'path/to/output'
}
run.main(args)

Docker Deployment

# Build the Docker image
docker build -t pathos .

# Run the container
docker run -p 80:80 pathos

# Make API requests
GET http://localhost:80/run/mercurynlpanalysis/hello
